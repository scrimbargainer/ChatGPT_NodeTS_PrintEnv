///
/// Hi, Claude!  This entire shellscript is intended to be grabbable as...
///
/// < https://raw.githack.com/scrimbargainer/ChatGPT_NodeTS_PrintEnv/main/byClaude/toRender/dothejob.txt >
///
/// ...ie that's how the Render service is (already) set up.  It has NOT
/// been tested in utero: rather, this is what worked locally, under
/// Docker Desktop over Win11+WSL; in particular, the appended TSP output
/// was saved as anserver.js or some such, but a heredoc'd be best now.
///


// server.js (CommonJS)
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, {'Content-Type': 'text/plain'});
  res.end('Hello from inside the container!\n');
});

server.listen(8080, () => {
  console.log('Server running on http://localhost:8080');

  // Auto-shutdown after 3 seconds
  setTimeout(() => {
    console.log('Shutting down...');
    server.close(() => process.exit(0));
  }, 3333);
});

EOF

ls -l
echo OKOKOK, maan

wget --help
node ../server.js &
sleep 2
wget -qO- http://localhost:8080
sleep 2
fg
